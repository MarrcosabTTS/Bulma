  <script>
    document.addEventListener('DOMContentLoaded', () => {
      const toggleButton = document.getElementById('js-toggle-skeleton');
      const statusLabel = document.getElementById('skeleton-status');
      
      let skeletonsActive = false;
      
      function toggleSkeletons() {
        // Seleccionar todos los elementos con clases is-skeleton o has-skeleton
        const skeletonElements = document.querySelectorAll('.is-skeleton, .has-skeleton, .skeleton-block, .skeleton-lines');
        
        skeletonElements.forEach((el) => {
          if (el.classList.contains('is-skeleton') || el.classList.contains('has-skeleton')) {
            // Alternar la clase skeleton para elementos estándar
            el.classList.toggle('is-skeleton');
            el.classList.toggle('has-skeleton');
          } else if (el.classList.contains('skeleton-block') || el.classList.contains('skeleton-lines')) {
            // Para los elementos especiales de bloque y líneas
            el.classList.toggle('is-skeleton');
          }
        });
      }
      
      toggleButton.addEventListener('click', () => {
        skeletonsActive = !skeletonsActive;
        
        statusLabel.textContent = skeletonsActive ? '(Active)' : '(Inactive)';
        toggleSkeletons();
      });
      
      // Inicializar el estado
      toggleSkeletons(); // Desactivar todos al inicio
      toggleSkeletons(); // Reactivar para que estén visibles inicialmente
    });
  </script>
